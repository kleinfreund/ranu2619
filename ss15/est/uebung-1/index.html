1.2 Winkelfunktionen – Wechselstromrechnung

c) Oszilloskopenschirm

\[\begin{align*} u(t = 0 \text{ ms}) &= 2 \text{ V}, u(t = 5.5 \text{ ms}) = 10.41 \text{ V} \\ \end{align*}\] \[\begin{align*} \omega &= 2 \pi f = 2 \pi \frac{1}{t} = \frac{2 \pi}{20 \text{ ms}} = \frac{\pi}{10 \text{ ms}} \\ \end{align*}\] \[\begin{align*} Û &= \frac{2 \text{ V}}{\sin(ω t + φ_{0})} \\ \end{align*}\] \[\begin{align*} Û &= \frac{10.41 \text{ V}}{\sin(ω t + φ_{0})} \\ \end{align*}\]
\[\begin{align*} \frac{2 \text{ V}}{\sin(ω t + φ_{0})} &= \frac{10.41 \text{ V}}{\sin(ω t + φ_{0})} \\ \frac{2 \text{ V}}{\sin(ω \cdot 0 \text{ ms} + φ_{0})} &= \frac{10.41 \text{ V}}{\sin(ω t + φ_{0})} \\ \Rightarrow \frac{2 \text{ V}}{\sin(φ_{0})} &= \frac{10.41 \text{ V}}{\sin(ω t + φ_{0})} ~| \cdot \sin(ω t + φ_{0})\\ \frac{2 \text{ V} \cdot \sin(ω t + φ_{0})}{\sin(φ_{0})} &= 10.41 \text{ V} ~| : 2 \text{ V} \\ \frac{\sin(ω t + φ_{0})}{\sin(φ_{0})} &= 5.205 \\ \Rightarrow \frac{\sin(ω t)\cos(φ_{0}) + \cos(ω t)\sin(φ_{0})}{\sin(φ_{0})} &= 5.205 \\ \frac{\sin(ω t)\cos(φ_{0})}{\sin(φ_{0})} + \cos(ω t) &= 5.205 ~| - \cos(ω t) \\ \frac{\sin(ω t)\cos(φ_{0})}{\sin(φ_{0})} &= 5.205 - \cos(ω t) ~| : \sin(ω t) \\ \frac{\cos(φ_{0})}{\sin(φ_{0})} &= \frac{5.205 - \cos(ω t)}{\sin(ω t)} \\ \cot(φ_{0}) &= \frac{5.205 - \cos(ω t)}{\sin(ω t)} \\ φ_{0} &= \text{arccot}\left( \frac{5.205 - \cos(ω t)}{\sin(ω t)} \right) \\ φ_{0} &= \text{arccot}\left( \frac{5.205 - \cos(\frac{ 5.5 \cdot \pi }{ 10 }) }{ \sin(\frac{ 5.5 \cdot \pi }{ 10 }) } \right) \\ φ_{0} &= \text{arccot}\left( 5.4282 \right) \\ φ_{0} &= 0.1822 \\ φ_{0} &= 10.4381 ° \\ \end{align*}\]
\[\begin{align*} Û &= \frac{ 2 \text{ V} }{ \sin(0.1822) } \\ Û &= 11.0392 \\ \end{align*}\] \[\begin{align*} Û &= \frac{ 10.41 \text{ V} }{ \sin(\frac{ 5.5 \cdot \pi }{ 10 } + 0.182) } \\ Û &= 11.0392 \\ \end{align*}\]
